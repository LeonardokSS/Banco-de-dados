As arrays funcionam como listas do python, podendo ser modificadas pelo índice



**Criando os alunos com disciplinas numa array
db.alunos.insertMany([

 "Programação"]

 "Sistemas Embarcados"]
 },

 ]
 }

//INSERIR $PUSH, quando inserimos com o $push, não verificamos a existência de outros dados iguais

db.alunos.updateOne(
    {_id : 1},
    {$push:{disciplinas:"Eletrônica 2"}})


//INSERIR $addToSet, Serve para iserir um novo dado dentro de uma array, verificando se não há duplicidade de dados (EVITA DUCPLICATA)

db.alunos.updateOne(
    {_id: 1},
    {$addToSet:{diciplinas:"Inovação e Tecnologia"}}
)


//DELETAR $Pull, remove algum dado pelo nome, então você não precisa saber o índice de algum elemento

db.alunos.updateOne(
{_id:1},
{$pull:{diciplinas:"Eletrônica 2"}})


                                        //**ÍNDICE**//

//REMOVER O ÚLTIMO ELEMENTO

db.alunos.updateOne(
    {_id:1},
    {$pop:{diciplinas : 1}}
)


//REMOVER O PRIMEIRO ELEMENTO

db.alunos.updateOne(
    {_id:1},
    {$pop:{diciplinas : -1}}
)



//FIND, procurar alunos com pelo menos 3 matérias pelo índice

db.alunos.find({ "disciplinas.3": { $exists: true } })

//FIND, verificar alunos que tem a diciplina Circuitos

db.alunos.find({ disciplinas: "Circuitos" })
